<h2>Конструктор: два способа инициализации</h2>

```c++
#include <iostream>

using namespace std;

struct Time {
    int days, hours, minutes, seconds;

    Time(int d, int h, int m, int s)
            : days(d), hours(h), minutes(m), seconds(s) {}
};

int main() {
    Time t(-3, 10, 1, 15);
    cout << t.days << ' ' << t.hours << ' ' << t.minutes << ' ' << t.seconds << endl;
    // печатает -3 10 1 15
    return 0;
}
```
